- # 下载vsftpd
	- ```bash
	  sudo apt install vsftpd
	  ```
- # 建立ftp专用用户和用户组
- # 配置vsftpd.conf
	- ```bash
	  sudo vim /etc/vsftpd.conf
	  ```
- # 启动vsftpd服务
	- ```bash
	  sudo systemctl service vsftpd enable
	  sudo systemctl service vsftpd restart
	  ```
